let simulator; const canvas = document.getElementById("renderCanvas"); const engine = new BABYLON.Engine(canvas, true); const createScene = function() { const scene = new BABYLON.Scene(engine); scene.clearColor = new BABYLON.Color3(0.5, 0.8, 0.9); // Camera const camera = new BABYLON.ArcRotateCamera("camera", Math.PI / 4, Math.PI / 3, 20, BABYLON.Vector3.Zero(), scene ); camera.attachControl(canvas, true); camera.lowerRadiusLimit = 10; camera.upperRadiusLimit = 50; // Light const light = new BABYLON.HemisphericLight("light", new BABYLON.Vector3(0, 1, 0), scene ); light.intensity = 0.7; const renderer = new BabylonRenderer(scene); simulator = new PhysicsSimulator(scene); // Create house const house = renderer.render({ type: "house", properties: { width: 8, height: 6, depth: 10, wallThickness: 0.3 } }); return scene; }; const scene = createScene(); engine.runRenderLoop(function() { scene.render(); }); window.addEventListener("resize", function() { engine.resize(); }); function startEarthquake() { simulator.startEarthquake(10000); } function stopEarthquake() { simulator.stopEarthquake(); } function updateEarthquakeIntensity() { const intensity = document.getElementById("intensitySlider").value; simulator.setEarthquakeIntensity(intensity, 2); }
